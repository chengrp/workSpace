## 交互流程模板

```python
# Claude 使用此模板进行交互

# 1. 内容分析
analysis = analyze_content(topic)
print(f"📊 内容分析：{analysis}")

# 2. 风格推荐
recommended = recommend_styles(analysis)
display_style_ranking(recommended)

# 3. 用户选择（强制停止点）
choice = AskUserQuestion(
    question="基于内容分析，为您推荐【XXX】。请选择：",
    options=[
        "🥇 使用推荐：XXX",
        "🥈 备选：YYY",
        "🥉 备选：ZZZ",
        "其他风格"
    ]
)

# 4. 规划模块
modules = plan_modules(topic, style)
display_modules(modules)

# 5. 生成 Prompt
prompt = generate_prompt(topic, modules, style)
filepath = save_prompt(topic, style, prompt)
print(f"📁 提示词已保存：{filepath}")

# 6. 确认生图
confirm = AskUserQuestion(
    question="是否开始生成图片？",
    options=["确认生成", "修改 Prompt"]
)

if confirm == "确认生成":
    image = generate_image(prompt)
    print(f"✅ 图片已生成：{image.path}")
```
